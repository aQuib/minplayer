  7 /**
  8  * @constructor

 10  * @class The base media player class where all media players derive from.
 11  *
 12  * @param {object} context The jQuery context.
 13  * @param {object} options This components options.
 14  * @param {object} mediaFile The media file for this player.
 15  */
 16 minplayer.players.base = function(context, options, mediaFile) {
 17

 50 /**
 51  * @see minplayer.plugin.construct
 52  * @this minplayer.players.base
 53  */
 54 minplayer.players.base.prototype.construct = function() {
 55 
 56   // Call the media display constructor.
 57   minplayer.display.prototype.construct.call(this);
 58 
 59   // Get the player display object.
 60   if (!this.playerFound()) {
 61 
 62     // Cleanup some events and code.
 63     this.display.unbind();
 64 
 65     // Remove the media element if found
 66     if (this.elements.media) {
 67       this.elements.media.remove();
 68     }
 69 
 70     // Create a new media player element.
 71     this.display.html(this.create());
 72   }
 73 
 74   // Get the player object...
 75   this.player = this.getPlayer();
 76 
 77   /**
 78    * Trigger a media event.
 79    * @this media.players.base.
 80    * @param {string} type The event type.
 81    * @param {object} data The event data object.
 82    */
 83   this.trigger = function(type, data) {
 84     this.display.trigger(type, data);
 85   };
 86 
 87   this.duration = 0;
 88   this.currentTime = 0;
 89 };
