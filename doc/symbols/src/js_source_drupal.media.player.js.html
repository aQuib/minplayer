 16   /**
 17    * @class The core media player class which governs the media player
 18    * functionality.
 19    *
 20    * <p><strong>Usage:</strong>
 21    * <pre><code>
 22    *
 23    *   // Create a media player.
 24    *   var player = $("#player").mediaplayer({
 25    *
 26    *   });
 27    *
 28    * </code></pre>
 29    * </p>
 30    *
 31    * @extends media.display
 32    * @param {object} context The jQuery context.
 33    * @param {object} options This components options.
 34    */
 35   media.player = function(context, options) {
 36 
 37     // Make sure we provide default options...
 38     options = $.extend({
 39       id: 'player',
 40       controller: 'default',
 41       template: 'default',
 42       volume: 80,
 43       swfplayer: '',
 44       wmode: 'transparent',
 45       attributes: {},
 46       settings: {}
 47     }, options);
 48 
 49     // Store this player instance.
 50     media.player[options.id] = this;
 51 
 52     // Derive from display
 53     media.display.call(this, context, options);
 54   };
 55 
 56   /**
 57    * Define the media player interface.
 58    */
 59   media.player.prototype = new media.display();
 60   media.player.prototype.constructor = media.player;
 61 
 62   /**
 63    * @see media.plugin.construct
 64    */
 65   media.player.prototype.construct = function() {
 66 
 67     // Call the media display constructor.
 68     media.display.prototype.construct.call(this);
 69 
 70     // Store the this pointer for callbacks.
 71     var pluginInfo = {};
 72     var plugin = null;
 73     var pluginContext = null;
 74     var i = media.plugins.length;
 75     var _files = [];
 76     var mediaSrc = null;
 77 
 78     /** The current player. */
 79     this.media = null;
 80 
 81     /** All of the plugin objects. */
 82     this.allPlugins = {};
 83 
 84     // Iterate through all of our plugins and add them to our plugins array.
 85     while (i--) {
 86       pluginInfo = media.plugins[i];
 87       if (pluginInfo.element) {
 88         pluginContext = $(pluginInfo.element, this.display);
 89       }
 90       else {
 91         pluginContext = this.display;
 92       }
 93       plugin = new pluginInfo.object(pluginContext, this.options);
 94       this.addPlugin(pluginInfo.id, plugin);
 95     }
 96 
 97     /** Variable to store the current media player. */
 98     this.currentPlayer = 'html5';
 99 
100     // Get the files involved...
101     if (this.elements.media) {
102       mediaSrc = this.elements.media.attr('src');
103       if (mediaSrc) {
104         _files.push({'path': mediaSrc});
105       }
106       $('source', this.elements.media).each(function() {
107         _files.push({
108           'path': $(this).attr('src'),
109           'mimetype': $(this).attr('type'),
110           'codecs': $(this).attr('codecs')
111         });
112       });
113     }
114 
115     // Add key events to the window.
116     $(window).bind('keyup', {obj: this}, function(event) {
117       // Escape out of fullscreen if they press ESC or Q.
118       var isFull = event.data.obj.display.hasClass('fullscreen');
119       if (isFull && (event.keyCode === 113 || event.keyCode === 27)) {
120         event.data.obj.display.removeClass('fullscreen');
121       }
122     });
123 
124     // Now load these files.
125     this.load(_files);
126   };
127 
128   /**
129    * Returns the full media player object.
130    * @param {array} files An array of files to chose from.
131    * @return {object} The best media file to play in the current browser.
132    */
133   media.player.prototype.getMediaFile = function(files) {
134     if (typeof files === 'string') {
135       return new media.file({'path': files});
136     }
137 
138     if (files.path) {
139       return new media.file(files);
140     }
141 
142     // Add the files and get the best player to play.
143     var i = files.length, bestPriority = 0, mFile = null, file = null;
144     while (i--) {
145       file = files[i];
146 
147       // Get the media file object.
148       if (typeof file === 'string') {
149         file = new media.file({'path': file});
150       }
151       else {
152         file = new media.file(file);
153       }
154 
155       // Determine the best file for this browser.
156       if (file.priority > bestPriority) {
157         mFile = file;
158       }
159     }
160 
161     // Return the media file.
162     return mFile;
163   };
164 
165   /**
166    * Load a set of files or a single file for the media player.
167    *
168    * @param {array} files An array of files to chose from to load.
169    */
170   media.player.prototype.load = function(files) {
171 
172     // Get the best media file.
173     var mFile = this.getMediaFile(files), id = '', pClass = '';
174 
175     // Only do something if we have a valid file.
176     if (mFile) {
177 
178       // Only destroy if the current player is different than the new player.
179       if (!this.media || (mFile.player.toString() !== this.currentPlayer)) {
180 
181         // Set the current media player.
182         this.currentPlayer = mFile.player.toString();
183 
184         // Make sure the player has an option to cleanup.
185         if (this.media) {
186           this.media.destroy();
187         }
188 
189         // Create a new media player for this file.
190         if (this.elements.display) {
191           pClass = media.players[mFile.player];
192           this.media = new pClass(this.elements.display, this.options, mFile);
193         }
194 
195         // Iterate through all plugins and add the player to them.
196         for (id in this.allPlugins) {
197           if (this.allPlugins.hasOwnProperty(id)) {
198             this.allPlugins[id].setPlayer(this.media);
199           }
200         }
201       }
202 
203       if (this.media) {
204         // Now load this media.
205         this.media.load(mFile);
206       }
207     }
208   };

210   /**
211    * Add a new plugin to the media player.
212    *
213    * @param {string} id The plugin ID.
214    * @param {object} plugin A new plugin object, derived from media.plugin.
215    */
216   media.player.prototype.addPlugin = function(id, plugin) {
217 
218     // Only continue if the plugin exists.
219     if (plugin.isValid()) {
220 
221       // Add to plugins.
222       this.allPlugins[id] = plugin;
223     }
224   };

226   /**
227    * Returns a plugin provided a plugin ID.
228    *
229    * @param {string} id The plugin ID to retrieve.
230    * @return {object} The plugin matching the provided ID.
231    */
232   media.player.prototype.getPlugin = function(id) {
233     return this.allPlugins[id];
234   };
