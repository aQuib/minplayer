1 /** The Drupal namespace. */
  2 Drupal.media = Drupal.media || {};
  3 (function($, media) {
  4 
  5   /** Define the playLoader object. */
  6   media.playLoader = media.playLoader || {};
  7 
  8   /**
  9    * @class The play loader base class, which is used to control the busy
 10    * cursor, big play button, and the opaque background which shows when the
 11    * player is paused.
 12    *
 13    * @extends media.display
 14    * @param {object} context The jQuery context.
 15    * @param {object} options This components options.
 16    */
 17   media.playLoader.base = function(context, options) {
 18 
 19     // Define the flags that control the busy cursor.
 20     this.busy = new media.flags();
 21 
 22     // Define the flags that control the big play button.
 23     this.bigPlay = new media.flags();
 24 
 25     // Derive from display
 26     media.display.call(this, context, options);
 27   };
 28 
 29   // Define the prototype for all controllers.
 30   media.playLoader.base.prototype = new media.display();
 31   media.playLoader.base.prototype.constructor = media.playLoader.base;
 32 
 33   /**
 34    * @see media.plugin#construct
 35    */
 36   media.playLoader.base.prototype.construct = function() {
 37 
 38     // Call the media plugin constructor.
 39     media.display.prototype.construct.call(this);
 40 
 41     // Trigger a play event when someone clicks on the controller.
 42     if (this.elements.bigPlay) {
 43       this.elements.bigPlay.bind('click', {obj: this}, function(event) {
 44         event.preventDefault();
 45         $(this).hide();
 46         if (event.data.obj.player) {
 47           event.data.obj.player.play();
 48         }
 49       });
 50     }
 51   };
 52 
 53   /**
 54    * Hide or show certain elements based on the state of the busy and big play
 55    * button.
 56    */
 57   media.playLoader.base.prototype.checkVisibility = function() {
 58 
 59     // Hide or show the busy cursor based on the flags.
 60     if (this.busy.flag) {
 61       this.elements.busy.show();
 62     }
 63     else {
 64       this.elements.busy.hide();
 65     }
 66 
 67     // Hide or show the big play button based on the flags.
 68     if (this.bigPlay.flag) {
 69       this.elements.bigPlay.show();
 70     }
 71     else {
 72       this.elements.bigPlay.hide();
 73     }
 74 
 75     // Show the control either flag is set.
 76     if (this.bigPlay.flag || this.busy.flag) {
 77       this.display.show();
 78     }
 79 
 80     // Hide the whole control if both flags are 0.
 81     if (!this.bigPlay.flag && !this.busy.flag) {
 82       this.display.hide();
 83     }
 84   };
 85 
 86   /**
 87    * @see media.plugin#setPlayer
 88    */
 89   media.playLoader.base.prototype.setPlayer = function(player) {
 90     media.display.prototype.setPlayer.call(this, player);
 91     var _this = this;
 92     player.display.bind('loadstart', function(event) {
 93       _this.busy.setFlag('media', true);
 94       _this.bigPlay.setFlag('media', true);
 95       _this.checkVisibility();
 96     });
 97     player.display.bind('waiting', function(event) {
 98       _this.busy.setFlag('media', true);
 99       _this.checkVisibility();
100     });
101     player.display.bind('loadedmetadata', function(event) {
102       _this.busy.setFlag('media', false);
103       _this.checkVisibility();
104     });
105     player.display.bind('loadeddata', function(event) {
106       _this.busy.setFlag('media', false);
107       _this.checkVisibility();
108     });
109     player.display.bind('playing', function(event) {
110       _this.busy.setFlag('media', false);
111       _this.bigPlay.setFlag('media', false);
112       _this.checkVisibility();
113     });
114     player.display.bind('pause', function(event) {
115       _this.bigPlay.setFlag('media', true);
116       _this.checkVisibility();
117     });
118   };
119 }(jQuery, Drupal.media));
