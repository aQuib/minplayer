1 /** The minplayer namespace. */
  2 minplayer = minplayer || {};
  3 
  4 /** Store all plugins. */
  5 minplayer.plugin_registry = {};
  6 
  7 /**
  8  * @constructor
  9  * @class The base class for all plugins.
 10  *
 11  * @param {object} context The jQuery context.
 12  * @param {object} options This components options.
 13  */
 14 minplayer.plugin = function(context, options) {
 15 
 16   // The media player.
 17   this.player = null;
 18 
 19   // Only call the constructor if we have a context.
 20   if (context) {
 21     this.construct();
 22   }
 23 };
 24 
 25 /**
 26  * The constructor which is called once the context is set.
 27  * Any class deriving from the plugin class should place all context
 28  * dependant functionality within this function instead of the standard
 29  * constructor function since it is called on object derivation as well
 30  * as object creation.
 31  */
 32 minplayer.plugin.prototype.construct = function() {
 33   // Add this plugin to the plugin registry.
 34   if (!minplayer.plugin_registry[this.options.id]) {
 35     minplayer.plugin_registry[this.options.id] = [];
 36   }
 37   minplayer.plugin_registry[this.options.id].push(this);
 38 };
 39 
 40 /**
 41  * Sets the current media player.
 42  *
 43  * @param {object} player The current media player.
 44  */
 45 minplayer.plugin.prototype.setPlayer = function(player) {
 46   this.player = player;
 47 };
