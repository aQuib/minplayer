1 /** The minplayer namespace. */
  2 minplayer = minplayer || {};
  3 
  4 /**
  5  * @constructor
  6  * @class The base class for all plugins.
  7  *
  8  * @param {object} context The jQuery context.
  9  * @param {object} options This components options.
 10  */
 11 minplayer.plugin = function(context, options) {
 12 
 13   // Only call the constructor if we have a context.
 14   if (context) {
 15     this.construct();
 16   }
 17 };
 18 
 19 /** Static array to keep track of plugin instances. */
 20 minplayer.plugin.instances = {};
 21 
 22 /**
 23  * The constructor which is called once the context is set.
 24  * Any class deriving from the plugin class should place all context
 25  * dependant functionality within this function instead of the standard
 26  * constructor function since it is called on object derivation as well
 27  * as object creation.
 28  */
 29 minplayer.plugin.prototype.construct = function() {
 30 
 31   // If this is a player, then it needs a new plugin
 32   if (this.options.name == 'player') {
 33     this.loadPlugins();
 34     this.options.name = 'player';
 35   }
 36 
 37   // Add this plugin.
 38   this.addPlugin(this.options.name, this);
 39 };
 40 
 41 /**
 42  * Destructor.
 43  */
 44 minplayer.plugin.prototype.destroy = function() {
 45   // Remove the pointer to the plugins array so it will be garbage collected.
 46   if (minplayer.plugin.instances[this.options.id][this.options.name]) {
 47     minplayer.plugin.instances[this.options.id][this.options.name] = null;
 48   }
 49 };
 50 
 51 /**
 52  * Adds a new plugin to this player.
 53  *
 54  * @param {string} name The name of this plugin.
 55  * @param {object} plugin A new plugin object, derived from media.plugin.
 56  */
 57 minplayer.plugin.prototype.addPlugin = function(name, plugin) {
 58 
 59   // Only continue if the plugin exists.
 60   if (plugin.isValid()) {
 61 
 62     // Add this to the plugins.
 63     minplayer.plugin.instances[this.options.id][name] = plugin;
 64   }
 65 };
 66 
 67 /**
 68  * Gets a plugin by name.
 69  *
 70  * @param {string} name The name of the plugin.
 71  * @return {object} The plugin for the provided name.
 72  */
 73 minplayer.plugin.prototype.getPlugin = function(name) {
 74   if (minplayer.plugin.instances[this.options.id][name]) {
 75     return minplayer.plugin.instances[this.options.id][name];
 76   }
 77   return null;
 78 };
 79 
 80 /**
 81  * Loads all of the available plugins.
 82  */
 83 minplayer.plugin.prototype.loadPlugins = function() {
 84   var plugin = null;
 85   var pluginInfo = {};
 86   var pluginContext = null;
 87 
 88   // Initialize this plugins array.
 89   minplayer.plugin.instances[this.options.id] = {};
 90 
 91   // Iterate through all the plugins.
 92   var i = minplayer.plugins.length;
 93   while (i--) {
 94 
 95     // Get the plugin information.
 96     pluginInfo = minplayer.plugins[i];
 97     if (pluginInfo.element) {
 98       pluginContext = jQuery(pluginInfo.element, this.display);
 99     }
100     else {
101       pluginContext = this.display;
102     }
103 
104     // Create the new plugin.
105     plugin = new pluginInfo.object(pluginContext, this.options);
106   }
107 };
108 
109 /**
110  * Iterate over each plugin.
111  *
112  * @param {function} callback Called for each plugin in this player.
113  */
114 minplayer.plugin.prototype.eachPlugin = function(callback) {
115   for (var name in minplayer.plugin.instances[this.options.id]) {
116     if (minplayer.plugin.instances[this.options.id].hasOwnProperty(name)) {
117       callback.call(this, minplayer.plugin.instances[this.options.id][name]);
118     }
119   }
120 };
121 
122 /**
123  * Initializes the plugin.
124  */
125 minplayer.plugin.prototype.initialize = function() {
126 };
