1 /** The minplayer namespace. */
  2 minplayer = minplayer || {};
  3 
  4 /**
  5  * @constructor
  6  * @class The base class for all plugins.
  7  *
  8  * @param {string} name The name of this plugin.
  9  * @param {object} context The jQuery context.
 10  * @param {object} options This components options.
 11  */
 12 minplayer.plugin = function(name, context, options) {
 13 
 14   // The name of this plugin.
 15   this.name = name;
 16 
 17   // Only call the constructor if we have a context.
 18   if (context) {
 19     this.construct();
 20   }
 21 };
 22 
 23 /**
 24  * The constructor which is called once the context is set.
 25  * Any class deriving from the plugin class should place all context
 26  * dependant functionality within this function instead of the standard
 27  * constructor function since it is called on object derivation as well
 28  * as object creation.
 29  */
 30 minplayer.plugin.prototype.construct = function() {
 31 
 32   // Adds this as a plugin.
 33   this.addPlugin();
 34 };
 35 
 36 /**
 37  * Destructor.
 38  */
 39 minplayer.plugin.prototype.destroy = function() {
 40   var plugin = this.getPlugin();
 41   plugin = null;
 42 };
 43 
 44 /**
 45  * Loads all of the available plugins.
 46  */
 47 minplayer.plugin.prototype.loadPlugins = function() {
 48   var plugin = null;
 49   var pluginInfo = {};
 50   var pluginContext = null;
 51 
 52   // Iterate through all the plugins.
 53   var i = minplayer.plugins.length;
 54   while (i--) {
 55 
 56     // Get the plugin information.
 57     pluginInfo = minplayer.plugins[i];
 58     if (pluginInfo.element) {
 59       pluginContext = jQuery(pluginInfo.element, this.display);
 60     }
 61     else {
 62       pluginContext = this.display;
 63     }
 64 
 65     // Create the new plugin.
 66     plugin = new pluginInfo.object(pluginContext, this.options);
 67   }
 68 };
 69 
 70 /**
 71  * Plugins should call this method when they are ready.
 72  */
 73 minplayer.plugin.prototype.ready = function() {
 74 
 75   // Set the loading flag.
 76   this.setLoading(false);
 77 
 78   // Check to see if all loading flags are 0.
 79   if (!this.loading()) {
 80 
 81     // Iterate through all plugins.
 82     this.eachPlugin(function(name, plugin) {
 83 
 84       // Initialize this plugin.
 85       plugin.initialize();
 86     });
 87   }
 88 };
 89 
 90 /**
 91  * Initializes the plugin.
 92  */
 93 minplayer.plugin.prototype.initialize = function() {
 94 };
 95 
 96 /**
 97  * Iterate over each plugin.
 98  *
 99  * @param {function} callback Called for each plugin in this player.
100  */
101 minplayer.plugin.prototype.eachPlugin = function(callback) {
102   var plugins = this.getPlugins();
103   for (var name in plugins) {
104     if (plugins.hasOwnProperty(name)) {
105       callback.call(this, name, plugins[name]);
106     }
107   }
108 };
109 
110 /**
111  * Adds a new plugin to this player.
112  *
113  * @param {string} name The name of this plugin.
114  * @param {object} plugin A new plugin object, derived from media.plugin.
115  */
116 minplayer.plugin.prototype.addPlugin = function(name, plugin) {
117   name = name || this.name;
118   plugin = plugin || this;
119 
120   // Make sure the plugin is valid.
121   if (plugin.isValid()) {
122 
123     // Set the plugin as loading.
124     plugin.setLoading(true);
125 
126     // Add the plugin to the plugins array.
127     this.getPlugins()[name] = plugin;
128   }
129 };
130 
131 /**
132  * Gets a plugin by name.
133  *
134  * @param {string} name The name of the plugin.
135  * @return {object} The plugin for the provided name.
136  */
137 minplayer.plugin.prototype.getPlugin = function(name) {
138   name = name || this.name;
139   return minplayer.plugin.get(this.options.id, name);
140 };
141 
142 /** Static array to keep track of plugin instances. */
143 minplayer.plugin.instances = {};
144 
145 /** Static variable to keep track of loading states for each widget. */
146 minplayer.plugin.loading = {};
147 
148 /**
149  * Iterate over each plugin instance.
150  *
151  * @param {string} type The type of plugin you wish to get.
152  * @param {function} callback Called for every plugin of this type.
153  */
154 minplayer.plugin.each = function(type, callback) {
155   for (var id in minplayer.plugin.instances) {
156     var instances = minplayer.plugin.instances[id];
157     for (var name in instances) {
158       if (name === type) {
159         callback.call(instances[name], instances[name]);
160       }
161     }
162   }
163 };
164 
165 /**
166  * Get a specific plugin.
167  *
168  * @param {string} id The ID of the widget to get the plugins from.
169  * @param {string} type The type of widget to get.
170  * @return {object} The plugin instance.
171  */
172 minplayer.plugin.get = function(id, type) {
173 
174   // If the plugins for this instance do not exist.
175   var instances = minplayer.plugin.instances[id];
176   if (instances && instances[type]) {
177     return instances[type];
178   }
179 
180   // Return this instance.
181   return null;
182 };
183 
184 /**
185  * Sets the loading flag.
186  *
187  * @param {boolean} state If this plugin is loading or not.
188  */
189 minplayer.plugin.prototype.setLoading = function(state) {
190   if (!minplayer.plugin.loading[this.options.id]) {
191     minplayer.plugin.loading[this.options.id] = new minplayer.flags();
192   }
193 
194   // Set this loading flag.
195   minplayer.plugin.loading[this.options.id].setFlag(this.name, state);
196 };
197 
198 /**
199  * Determine if this widget is still loading.
200  *
201  * @return {number} 0 => Widget is done loading, >0 => Widget is still loading.
202  */
203 minplayer.plugin.prototype.loading = function() {
204   return minplayer.plugin.loading[this.options.id].flag;
205 };
206 
207 /**
208  * Returns the plugins for this ID.
209  *
210  * @return {array} An array of plugins.
211  */
212 minplayer.plugin.prototype.getPlugins = function() {
213 
214   // If the plugins for this instance do not exist.
215   if (!minplayer.plugin.instances[this.options.id]) {
216 
217     // Initialize the instances.
218     minplayer.plugin.instances[this.options.id] = {};
219 
220     // Now load all plugins.
221     this.loadPlugins();
222   }
223 
224   // Return the plugins for this instance.
225   return minplayer.plugin.instances[this.options.id];
226 };
