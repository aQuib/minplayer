  7 /**
  8  * @constructor
  9  * @extends minplayer.display
 10  * @class The Flash media player class to control the flash fallback.
 11  *
 12  * @param {object} context The jQuery context.
 13  * @param {object} options This components options.
 14  * @param {object} mediaFile The media file for this player.
 15  */
 16 minplayer.players.youtube = function(context, options, mediaFile) {
 17 
 18   // Derive from players flash.
 19   minplayer.players.flash.call(this, context, options, mediaFile);
 20 
 21   /** The quality of the YouTube stream. */
 22   this.quality = 'default';
 23 };

 25 /** Derive from minplayer.players.flash. */
 26 minplayer.players.youtube.prototype = new minplayer.players.flash();
 27 
 28 /** Reset the constructor. */
 29 minplayer.players.youtube.prototype.constructor = minplayer.players.youtube;

 31 /**
 32  * Called when the YouTube player is ready.
 33  *
 34  * @param {string} id The media player ID.

 36 window.onYouTubePlayerReady = function(id) {
 37   if (minplayer.player[id]) {
 38     minplayer.player[id].media.onReady();
 39   }

 42 /**
 43  * @see minplayer.plugin.construct
 44  */
 45 minplayer.players.youtube.prototype.construct = function() {
 46 
 47   // Call flash constructor.
 48   minplayer.players.flash.prototype.construct.call(this);
 49 
 50   // Translates the player state for the YouTube API player.
 51   this.getPlayerState = function(playerState) {
 52     switch (playerState) {
 53       case 5:
 54         return 'ready';
 55       case 3:
 56         return 'waiting';
 57       case 2:
 58         return 'pause';
 59       case 1:
 60         return 'play';
 61       case 0:
 62         return 'ended';
 63       case -1:
 64         return 'abort';
 65       default:
 66         return 'unknown';
 67     }
 68   };
 69 
 70   // Create our callback functions.
 71   var _this = this;
 72   window[this.options.id + 'StateChange'] = function(newState) {
 73     _this.trigger(_this.getPlayerState(newState));
 74   };
 75 
 76   window[this.options.id + 'PlayerError'] = function(errorCode) {
 77     _this.trigger('error', errorCode);
 78   };
 79 
 80   window[this.options.id + 'QualityChange'] = function(newQuality) {
 81     _this.quality = newQuality;
 82   };
 83 
 84   // Add our event listeners.
 85   if (this.player) {
 86     var onStateChange = this.options.id + 'StateChange';
 87     var onError = this.options.id + 'PlayerError';
 88     var onQuality = this.options.id + 'QualityChange';
 89     this.player.addEventListener('onStateChange', onStateChange);
 90     this.player.addEventListener('onError', onError);
 91     this.player.addEventListener('onPlaybackQualityChange', onQuality);
 92   }
 93 };

103 /**
104  * @see minplayer.players.base#canPlay
105  * @param {object} file A {@link minplayer.file} object.
106  * @return {boolean} If this player can play this media type.
107  */
108 minplayer.players.youtube.canPlay = function(file) {
109 
110   // Check for the mimetype for youtube.
111   if (file.mimetype === 'video/youtube') {
112     return true;
113   }
114 
115   // If the path is a YouTube path, then return true.
116   return (file.path.search(/^http(s)?\:\/\/(www\.)?youtube\.com/i) === 0);
117 };

119 /**
120  * @see minplayer.players.base#create
121  * @return {object} The media player entity.
122  */
123 minplayer.players.youtube.prototype.create = function() {
124 
125   minplayer.players.base.prototype.flash.call(this);
126 
127   // The flash variables for this flash player.
128   var flashVars = {
129     'file': this.mediaFile.path,
130     'autostart': this.options.settings.autoplay
131   };
132 
133   // Return a flash media player object.
134   var rand = Math.floor(Math.random() * 1000000);
135   var flashPlayer = 'http://www.youtube.com/apiplayer?rand=' + rand;
136   flashPlayer += '&version=3&enablejsapi=1&playerapiid=';
137   flashPlayer += this.options.id;
138   return minplayer.players.flash.getFlash({
139     swf: flashPlayer,
140     id: this.options.id + '_player',
141     playerType: 'flash',
142     width: this.options.settings.width,
143     height: '100%',
144     flashvars: flashVars,
145     wmode: this.options.wmode
146   });
147 };

149 /**
150  * Return the Regular Expression to find a YouTube ID.
151  *
152  * @return {RegEx} A regular expression to find a YouTube ID.
153  */
154 minplayer.players.youtube.prototype.regex = function() {
155   return /^http[s]?\:\/\/(www\.)?youtube\.com\/watch\?v=([a-zA-Z0-9]+)/i;
156 };

158 /**
159  * @see minplayer.players.base#load
160  */
161 minplayer.players.youtube.prototype.load = function(file) {
162   minplayer.players.flash.prototype.load.call(this, file);
163   if (this.isReady()) {
164     var regex = this.regex();
165     var id = '';
166     if (this.mediaFile.path.search(regex) === 0) {
167       id = this.mediaFile.path.replace(regex, '$2');
168     }
169     else {
170       id = this.mediaFile.path;
171     }
172     this.player.loadVideoById(id, 0, this.quality);
173   }
174 };
