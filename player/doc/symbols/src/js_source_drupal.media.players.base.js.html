1 /** The Drupal namespace. */
  2 Drupal.media = Drupal.media || {};
  3 (function($, media) {
  4 
  5   /** All the media player implementations */
  6   media.players = media.players || {};
  7 
  8   /**
  9    * @class The base media player class where all media players derive from.
 10    *
 11    * @extends media.display
 12    * @param {object} context The jQuery context.
 13    * @param {object} options This components options.
 14    */
 15   media.players.base = function(context, options, mediaFile) {
 16 
 17     /** The currently loaded media file. */
 18     this.mediaFile = mediaFile;
 19 
 20     // Derive from display
 21     media.display.call(this, context, options);
 22   };
 23 
 24   /**
 25    * Get the priority of this media player.
 26    *
 27    * @return {number} The priority of this media player.
 28    */
 29   media.players.base.getPriority = function() {
 30     return 0;
 31   };
 32 
 33   /**
 34    * Determine if we can play the media file.
 35    *
 36    * @param {object} file A {@link media.file} object.
 37    */
 38   media.players.base.canPlay = function(file) {
 39     return false;
 40   };
 41 
 42   // Extend the display prototype.
 43   media.players.base.prototype = new media.display();
 44   media.players.base.prototype.constructor = media.players.base;
 45 
 46   /**
 47    * @see media.plugin.construct
 48    */
 49   media.players.base.prototype.construct = function() {
 50 
 51     // Call the media display constructor.
 52     media.display.prototype.construct.call(this);
 53 
 54     // Get the player display object.
 55     if (!this.playerFound()) {
 56 
 57       // Cleanup some events and code.
 58       this.display.unbind();
 59 
 60       // Remove the media element if found
 61       if (this.elements.media) {
 62         this.elements.media.remove();
 63       }
 64 
 65       // Create a new media player element.
 66       this.display.html(this.create());
 67     }
 68 
 69     // Get the player object...
 70     this.player = this.getPlayer();
 71 
 72     /**
 73      * Trigger a media event.
 74      *
 75      * @param {string} type The event type.
 76      * @param {object} data The event data object.
 77      */
 78     this.trigger = function(type, data) {
 79       this.display.trigger(type, data);
 80     };
 81 
 82     this.duration = 0;
 83     this.currentTime = 0;
 84   };
 85 
 86   /**
 87    * Returns if the media player is already within the DOM.
 88    *
 89    * @return {boolean} TRUE - if the player is in the DOM, FALSE otherwise.
 90    */
 91   media.players.base.prototype.playerFound = function() {
 92     return false;
 93   };
 94 
 95   /**
 96    * Creates the media player and inserts it in the DOM.
 97    *
 98    * @return {object} The media player entity.
 99    */
100   media.players.base.prototype.create = function() {
101     return null;
102   };
103 
104   /**
105    * Returns the media player object.
106    *
107    * @return {object} The media player object.
108    */
109   media.players.base.prototype.getPlayer = function() {
110     return null;
111   };
112 
113   /**
114    * Destroy the media player instance from the DOM.
115    */
116   media.players.base.prototype.destroy = function() {
117   };
118 
119   /**
120    * Loads a new media player.
121    *
122    * @param {object} file A {@link media.file} object.
123    */
124   media.players.base.prototype.load = function(file) {
125 
126     // Store the media file for future lookup.
127     this.mediaFile = file;
128   };
129

131    * Play the loaded media file.

133   media.players.base.prototype.play = function() {

139   media.players.base.prototype.pause = function() {
140   };

146   };

150    *

152    */
153   media.players.base.prototype.seek = function(pos) {
154 
155   };
156 
157   /**
158    * Set the volume of the loaded media.
159    *
160    * @param {number} vol The volume to set the media. 0 to 1.
161    */
162   media.players.base.prototype.setVolume = function(vol) {
163     this.trigger('volumeupdate', vol);
164   };

166   /**
167    * Get the volume from the loaded media.
168    *

170    */
171   media.players.base.prototype.getVolume = function() {
172     return 0;
173   };

176    * Return the duration of the loaded media.
177    *
178    * @return {number} The duration of the loaded media.
179    */
180   media.players.base.prototype.getDuration = function() {
181     return 0;
