1 /** The Drupal namespace. */
  2 Drupal.media = Drupal.media || {};
  3 (function($, media) {
  4 
  5   /** All the media player implementations */
  6   media.players = media.players || {};
  7 
  8   /**
  9    * @class The Flash media player class to control the flash fallback.
 10    *
 11    * @extends media.display
 12    * @param {object} context The jQuery context.
 13    * @param {object} options This components options.
 14    */
 15   media.players.youtube = function(context, options, mediaFile) {
 16 
 17     // Derive from players flash.
 18     media.players.flash.call(this, context, options, mediaFile);
 19 
 20     /** The quality of the YouTube stream. */
 21     this.quality = 'default';
 22   };
 23 
 24   // Called when the YouTube player is ready.
 25   window.onYouTubePlayerReady = function(id) {
 26     if (media.player[id]) {
 27       media.player[id].media.onReady();
 28     }
 29   };
 30 
 31   /**
 32    * @see media.plugin.construct
 33    */
 34   media.players.youtube.prototype.construct = function() {
 35 
 36     // Call flash constructor.
 37     media.players.flash.prototype.construct.call(this);
 38 
 39     // Translates the player state for the YouTube API player.
 40     this.getPlayerState = function(playerState) {
 41       switch (playerState) {
 42         case 5:
 43           return 'ready';
 44         case 3:
 45           return 'waiting';
 46         case 2:
 47           return 'pause';
 48         case 1:
 49           return 'play';
 50         case 0:
 51           return 'ended';
 52         case -1:
 53           return 'abort';
 54         default:
 55           return 'unknown';
 56       }
 57     };
 58 
 59     // Create our callback functions.
 60     var _this = this;
 61     window[this.options.id + 'StateChange'] = function(newState) {
 62       _this.trigger(_this.getPlayerState(newState));
 63     };
 64 
 65     window[this.options.id + 'PlayerError'] = function(errorCode) {
 66       _this.trigger('error', errorCode);
 67     };
 68 
 69     window[this.options.id + 'QualityChange'] = function(newQuality) {
 70       _this.quality = newQuality;
 71     };
 72 
 73     // Add our event listeners.
 74     if (this.player) {
 75       var onStateChange = this.options.id + 'StateChange';
 76       var onError = this.options.id + 'PlayerError';
 77       var onQuality = this.options.id + 'QualityChange';
 78       this.player.addEventListener('onStateChange', onStateChange);
 79       this.player.addEventListener('onError', onError);
 80       this.player.addEventListener('onPlaybackQualityChange', onQuality);
 81     }
 82   };
 83 
 84   /**
 85    * @see media.players.base#getPriority
 86    */
 87   media.players.youtube.getPriority = function() {
 88     return 10;
 89   };
 90 
 91   /**
 92    * @see media.players.base#canPlay
 93    */
 94   media.players.youtube.canPlay = function(file) {
 95 
 96     // Check for the mimetype for youtube.
 97     if (file.mimetype === 'video/youtube') {
 98       return true;
 99     }
100 
101     // If the path is a YouTube path, then return true.
102     return (file.path.search(/^http(s)?\:\/\/(www\.)?youtube\.com/i) === 0);
103   };
104 
105   // Define the prototype.
106   media.players.youtube.prototype = new media.players.flash();
107   media.players.youtube.prototype.constructor = media.players.youtube;
108 
109   /**
110    * @see media.players.base#create
111    */
112   media.players.youtube.prototype.create = function() {
113 
114     media.players.base.prototype.flash.call(this);
115 
116     // The flash variables for this flash player.
117     var flashVars = {
118       'file': this.mediaFile.path,
119       'autostart': this.options.settings.autoplay
120     };
121 
122     // Return a flash media player object.
123     var rand = Math.floor(Math.random() * 1000000);
124     var flashPlayer = 'http://www.youtube.com/apiplayer?rand=' + rand;
125     flashPlayer += '&version=3&enablejsapi=1&playerapiid=';
126     flashPlayer += this.options.id;
127     return media.players.youtube.getFlash({
128       swf: flashPlayer,
129       id: this.options.id + '_player',
130       playerType: 'flash',
131       width: this.options.settings.width,
132       height: '100%',
133       flashvars: flashVars,
134       wmode: this.options.wmode
135     });
136   };
137 
138   /**
139    * Return the Regular Expression to find a YouTube ID.
140    *
141    * @return {RegEx} A regular expression to find a YouTube ID.
142    */
143   media.players.youtube.prototype.regex = function() {
144     return /^http[s]?\:\/\/(www\.)?youtube\.com\/watch\?v=([a-zA-Z0-9]+)/i;
145   };
146 
147   /**
148    * @see media.players.base#load
149    */
150   media.players.youtube.prototype.load = function(file) {
151     media.players.flash.prototype.load.call(this, file);
152     if (this.isReady()) {
153       var regex = this.regex();
154       var id = '';
155       if (this.mediaFile.path.search(regex) === 0) {
156         id = this.mediaFile.path.replace(regex, '$2');
157       }
158       else {
159         id = this.mediaFile.path;
160       }
161       this.player.loadVideoById(id, 0, this.quality);
162     }
163   };
164

166    * @see media.players.base#play
167    */
168   media.players.youtube.prototype.play = function() {
169     media.players.flash.prototype.play.call(this);
170     if (this.isReady()) {
171       this.player.playVideo();
172     }
173   };
