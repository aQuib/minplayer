1 /** The Drupal namespace. */
  2 Drupal.media = Drupal.media || {};
  3 (function($, media) {
  4 
  5   /** All the media player implementations */
  6   media.players = media.players || {};
  7 
  8   /**
  9    * @class The HTML5 media player implementation.
 10    *
 11    * @extends media.display
 12    * @param {object} context The jQuery context.
 13    * @param {object} options This components options.
 14    */
 15   media.players.html5 = function(context, options, mediaFile) {
 16 
 17     // Derive players base.
 18     media.players.base.call(this, context, options, mediaFile);
 19   };
 20 
 21   /**
 22    * @see media.players.base#getPriority
 23    */
 24   media.players.html5.getPriority = function() {
 25     return 10;
 26   };
 27 
 28   /**
 29    * @see media.players.base#canPlay
 30    */
 31   media.players.html5.canPlay = function(file) {
 32     switch (file.mimetype) {
 33       case 'video/ogg':
 34         return media.playTypes.videoOGG;
 35       case 'video/mp4':
 36         return media.playTypes.videoH264;
 37       case 'video/x-webm':
 38         return media.playTypes.videoWEBM;
 39       case 'audio/ogg':
 40         return media.playTypes.audioOGG;
 41       case 'audio/mpeg':
 42         return media.playTypes.audioMP3;
 43       case 'audio/mp4':
 44         return media.playTypes.audioMP4;
 45       default:
 46         return false;
 47     }
 48   };
 49 
 50   // Define the prototype.
 51   media.players.html5.prototype = new media.players.base();
 52   media.players.html5.prototype.constructor = media.players.html5;
 53 
 54   /**
 55    * @see media.plugin.construct
 56    */
 57   media.players.html5.prototype.construct = function() {
 58 
 59     // Call base constructor.
 60     media.players.base.prototype.construct.call(this);
 61 
 62     // See if we are loaded.
 63     this.loaded = false;
 64 
 65     // Store the this pointer...
 66     var _this = this;
 67 
 68     // For the HTML5 player, we will just pass events along...
 69     if (this.player) {
 70       this.player.addEventListener('abort', function() {
 71         _this.trigger('abort');
 72       }, true);
 73       this.player.addEventListener('loadstart', function() {
 74         _this.trigger('loadstart');
 75       }, true);
 76       this.player.addEventListener('loadeddata', function() {
 77         _this.trigger('loadeddata');
 78       }, true);
 79       this.player.addEventListener('loadedmetadata', function() {
 80         _this.trigger('loadedmetadata');
 81       }, true);
 82       this.player.addEventListener('canplaythrough', function() {
 83         _this.trigger('canplaythrough');
 84       }, true);
 85       this.player.addEventListener('ended', function() {
 86         _this.trigger('ended');
 87       }, true);
 88       this.player.addEventListener('pause', function() {
 89         _this.trigger('pause');
 90       }, true);
 91       this.player.addEventListener('play', function() {
 92         _this.trigger('play');
 93       }, true);
 94       this.player.addEventListener('playing', function() {
 95         _this.trigger('playing');
 96       }, true);
 97       this.player.addEventListener('error', function() {
 98         _this.trigger('error');
 99       }, true);
100       this.player.addEventListener('waiting', function() {
101         _this.trigger('waiting');
102       }, true);
103       this.player.addEventListener('timeupdate', function(event) {
104         var dur = this.duration;
105         var cTime = this.currentTime;
106         _this.duration = dur;
107         _this.currentTime = cTime;
108         _this.trigger('timeupdate', {currentTime: cTime, duration: dur});
109       }, true);
110       this.player.addEventListener('durationchange', function() {
111         _this.duration = this.duration;
112         _this.trigger('durationchange', {duration: this.duration});
113       }, true);
114       this.player.addEventListener('progress', function(event) {
115         _this.trigger('progress', {loaded: event.loaded, total: event.total});
116       }, true);
117 
118       if (this.autoBuffer()) {
119         this.player.autobuffer = true;
120       } else {
121         this.player.autobuffer = false;
122         this.player.preload = 'none';
123       }
124     }
125   };
126 
127   // Determine if this player is able to autobuffer.
128   media.players.html5.prototype.autoBuffer = function() {
129     var preload = this.player.preload !== 'none';
130     if (typeof this.player.hasAttribute === 'function') {
131       return this.player.hasAttribute('preload') && preload;
132     }
133     else {
134       return false;
135     }
136   };
137 
138   /**
139    * @see media.players.base#playerFound
140    */
141   media.players.html5.prototype.playerFound = function() {
142     return (this.display.find(this.mediaFile.type).length > 0);
143   };
144 
145   /**
146    * @see media.players.base#create
147    */
148   media.players.html5.prototype.create = function() {
149     var element = document.createElement(this.mediaFile.type), attribute = '';
150     for (attribute in this.options.attributes) {
151       if (this.options.attributes.hasOwnProperty(attribute)) {
152         element.setAttribute(attribute, this.options.attributes[attribute]);
153       }
154     }
155     return element;
156   };
157 
158   /**
159    * @see media.players.base#getPlayer
160    */
161   media.players.html5.prototype.getPlayer = function() {
162     return this.options.elements.media.eq(0)[0];
163   };
164 
165   /**
166    * @see media.players.base#load
167    */
168   media.players.html5.prototype.load = function(file) {
169     // Always call the base first on load...
170     media.players.base.prototype.load.call(this, file);
171 
172     if (this.loaded) {
173       // Change the source...
174       var code = '<source src="' + file.path + '" ';
175       code += 'type="' + file.mimetype + '"';
176       code += file.codecs ? ' codecs="' + file.path + '">' : '>';
177       this.options.elements.player.attr('src', '').empty().html(code);
178     }
179 
180     // Set the loaded flag.
181     this.loaded = true;
182   };
183 
184   /**
185    * @see media.players.base#play
186    */
187   media.players.html5.prototype.play = function() {
188     media.players.base.prototype.play.call(this);
189     this.player.play();
190   };
191 
192   /**
193    * @see media.players.base#pause
194    */
195   media.players.html5.prototype.pause = function() {
196     media.players.base.prototype.pause.call(this);
197     this.player.pause();
198   };
199 
200   /**
201    * @see media.players.base#stop
202    */
203   media.players.html5.prototype.stop = function() {
204     media.players.base.prototype.stop.call(this);
205     this.media.pause();
206     this.player.src = '';
207   };
208 
209   /**
210    * @see media.players.base#seek
211    */
212   media.players.html5.prototype.seek = function(pos) {
213     media.players.base.prototype.seek.call(this, pos);
214     this.player.currentTime = pos;
215   };
216 
217   /**
218    * @see media.players.base#setVolume
219    */
220   media.players.html5.prototype.setVolume = function(vol) {
221     media.players.base.prototype.setVolume.call(this, vol);
222     this.player.volume = vol;
223   };
224 
225   /**
226    * @see media.players.base#getVolume
227    */
228   media.players.html5.prototype.getVolume = function() {
229     return this.player.volume;
230   };
231 
232   /**
233    * @see media.players.base#getDuration
234    */
235   media.players.html5.prototype.getDuration = function() {
236     var dur = this.player.duration;
237     return (dur === Infinity) ? 0 : dur;
238   };
239 }(jQuery, Drupal.media));
