1 /** The Drupal namespace. */
  2 Drupal.media = Drupal.media || {};
  3 (function($, media) {
  4 
  5   /** Define the controllers object. */
  6   media.controllers = media.controllers || {};
  7 
  8   /**
  9    * @class This is the base media controller.  Other controllers can derive
 10    * from the base and either build on top of it or simply define the elements
 11    * that this base controller uses.
 12    *
 13    * @extends media.display
 14    * @param {object} context The jQuery context.
 15    * @param {object} options This components options.
 16    */
 17   media.controllers.base = function(context, options) {
 18 
 19     // Derive from display
 20     media.display.call(this, context, options);
 21   }
 22 
 23   /**
 24    * A static function that will format a time value into a string time format.
 25    *
 26    * @param {integer} time An integer value of time.
 27    * @return {string} A string representation of the time.
 28    */
 29   media.formatTime = function(time) {
 30     time = time || 0;
 31     var seconds = 0, minutes = 0, hour = 0, timeString = '';
 32 
 33     hour = Math.floor(time / 3600);
 34     time -= (hour * 3600);
 35     minutes = Math.floor(time / 60);
 36     time -= (minutes * 60);
 37     seconds = Math.floor(time % 60);
 38 
 39     if (hour) {
 40       timeString += String(hour);
 41       timeString += ':';
 42     }
 43 
 44     timeString += (minutes >= 10) ? String(minutes) : ('0' + String(minutes));
 45     timeString += ':';
 46     timeString += (seconds >= 10) ? String(seconds) : ('0' + String(seconds));
 47     return {time: timeString, units: ''};
 48   };
 49 
 50   // Define the prototype for all controllers.
 51   var controllersBase = media.controllers.base;
 52   media.controllers.base.prototype = new media.display();
 53   media.controllers.base.prototype.constructor = media.controllers.base;
 54 
 55   /**
 56    * @see media.display#getElements
 57    */
 58   media.controllers.base.prototype.getElements = function() {
 59     var elements = media.display.prototype.getElements.call(this);
 60     return $.extend(elements, {
 61       play: null,
 62       pause: null,
 63       fullscreen: null,
 64       seek: null,
 65       volume: null,
 66       timer: null
 67     });
 68   };
 69 
 70   /**
 71    * @see media.plugin#construct
 72    */
 73   media.controllers.base.prototype.construct = function() {
 74 
 75     // Call the media plugin constructor.
 76     media.display.prototype.construct.call(this);
 77 
 78     // Play or pause the player.
 79     function playPause(controller, state) {
 80       var type = state ? 'play' : 'pause';
 81       controller.display.trigger(type);
 82       controller.setPlayPause(state);
 83       if (controller.player) {
 84         controller.player[type]();
 85       }
 86     }
 87 
 88     // Trigger the controller events.
 89     if (this.elements.play) {
 90       this.elements.play.bind('click', {obj: this}, function(event) {
 91         event.preventDefault();
 92         playPause(event.data.obj, true);
 93       });
 94     }
 95 
 96     if (this.elements.pause) {
 97       this.elements.pause.bind('click', {obj: this}, function(event) {
 98         event.preventDefault();
 99         playPause(event.data.obj, false);
100       });
101     }
102 
103     // Fullscreen button.
104     var _this = this, sliderOptions = {};
105     if (this.elements.fullscreen) {
106       this.elements.fullscreen.click(function() {
107         var isFull = _this.elements.player.hasClass('fullscreen');
108         if (isFull) {
109           _this.elements.player.removeClass('fullscreen');
110         }
111         else {
112           _this.elements.player.addClass('fullscreen');
113         }
114       }).css({'pointer' : 'hand'});
115     }
116 
117     // Create the slider.
118     this.dragging = false;
119 
120     // Add a seekBar and volumeBar using jQuery UI Slider.
121     if (this.elements.seek) {
122       this.seekBar = this.elements.seek.slider({range: 'min'});
123     }
124     if (this.elements.volume) {
125       sliderOptions = {range: 'min', orientation: 'vertical'};
126       this.volumeBar = this.elements.volume.slider(sliderOptions);
127     }
128   };
129 
130   /**
131    * Sets the play and pause state of the control bar.
132    *
133    * @param {boolean} state TRUE - Show Play, FALSE - Show Pause.
134    */
135   media.controllers.base.prototype.setPlayPause = function(state) {
136     var css = '';
137     if (this.elements.play) {
138       css = state ? 'inherit' : 'none';
139       this.elements.play.css('display', css);
140     }
141     if (this.elements.pause) {
142       css = state ? 'none' : 'inherit';
143       this.elements.pause.css('display', css);
144     }
145   };
146 
147   /**
148    * Sets the time string on the control bar.
149    *
150    * @param {string} element The name of the element to set.
151    * @param {number} time The total time amount to set.
152    */
153   media.controllers.base.prototype.setTimeString = function(element, time) {
154     if (this.elements[element]) {
155       this.elements[element].text(media.formatTime(time).time);
156     }
157   };
158 
159   /**
160    * @see media.plugin#setPlayer
161    */
162   media.controllers.base.prototype.setPlayer = function(player) {
163     media.display.prototype.setPlayer.call(this, player);
164     player.display.bind('pause', {obj: this}, function(event) {
165       event.data.obj.setPlayPause(true);
166       clearInterval(event.data.obj.interval);
167     });
168     player.display.bind('playing', {obj: this}, function(event) {
169       event.data.obj.setPlayPause(false);
170     });
171     player.display.bind('durationchange', {obj: this}, function(event, data) {
172       event.data.obj.setTimeString('duration', data.duration);
173     });
174     player.display.bind('timeupdate', {obj: this}, function(event, data) {
175       if (!event.data.obj.dragging) {
176         var value = 0;
177         if (data.duration) {
178           value = (data.currentTime / data.duration) * 100;
179         }
180 
181         // Update the seek bar if it exists.
182         if (event.data.obj.seekBar) {
183           event.data.obj.seekBar.slider('option', 'value', value);
184         }
185 
186         event.data.obj.setTimeString('timer', data.currentTime);
187       }
188     });
189 
190     // Register the events for the control bar to control the media.
191     if (this.seekBar) {
192       var _this = this;
193       this.seekBar.slider({
194         start: function(event, ui) {
195           _this.dragging = true;
196         },
197         stop: function(event, ui) {
198           _this.dragging = false;
199           var time = (ui.value / 100) * player.getDuration();
200           player.seek(time);
201         },
202         slide: function(event, ui) {
203           var time = (ui.value / 100) * player.getDuration();
204           if (!_this.dragging) {
205             player.seek(time);
206           }
207 
208           _this.setTimeString('timer', time);
209         }
210       });
211     }
212 
213     // Register the volume bar to adjust the player volume.
214     player.setVolume(this.options.volume / 100);
215 
216     // Setup the volume bar.
217     if (this.volumeBar) {
218       this.volumeBar.slider('option', 'value', this.options.volume);
219       this.volumeBar.slider({
220         slide: function(event, ui) {
221           player.setVolume(ui.value / 100);
222         }
223       });
224     }
225   };
226 }(jQuery, Drupal.media));
