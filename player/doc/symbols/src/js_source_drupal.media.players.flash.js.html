  8   /**
  9    * @class The Flash media player class to control the flash fallback.
 10    *
 11    * @extends media.display
 12    * @param {object} context The jQuery context.
 13    * @param {object} options This components options.
 14    */
 15   media.players.flash = function(context, options, mediaFile) {
 16 
 17     this.durationInterval = null;
 18     this.mediaInterval = null;
 19     this.duration = 0;
 20     this.ready = false;
 21 
 22     // Derive from players base.
 23     media.players.base.call(this, context, options, mediaFile);
 24   };

 44   /**
 45    * API to return the Flash player code provided params.
 46    *
 47    * @param {object} params The params used to populate the Flash code.
 48    */
 49   media.players.flash.getFlash = function(params) {
 50     // Get the protocol.
 51     var protocol = window.location.protocol;
 52     var element = null;
 53     var embed = null;
 54     var paramKey = '';
 55     var flashParams = {};
 56     var param = null;
 57 
 58     if (protocol.charAt(protocol.length - 1) === ':') {
 59       protocol = protocol.substring(0, protocol.length - 1);
 60     }
 61 
 62     // Create an object element.
 63     element = document.createElement('object');
 64     element.setAttribute('width', params.width);
 65     element.setAttribute('height', params.height);
 66     element.setAttribute('id', params.id);
 67     element.setAttribute('name', params.id);
 68     element.setAttribute('playerType', params.playerType);
 69 
 70     // Setup a params array to make the param additions eaiser.
 71     flashParams = {
 72       'allowScriptAccess': 'always',
 73       'allowfullscreen': 'true',
 74       'movie': params.swf,
 75       'wmode': params.wmode,
 76       'quality': 'high',
 77       'FlashVars': $.param(params.flashvars)
 78     };
 79 
 80     // Add the parameters.
 81     for (paramKey in flashParams) {
 82       if (flashParams.hasOwnProperty(paramKey)) {
 83         param = document.createElement('param');
 84         param.setAttribute('name', paramKey);
 85         param.setAttribute('value', flashParams[paramKey]);
 86         element.appendChild(param);
 87       }
 88     }
 89 
 90     // Add the embed element.
 91     embed = document.createElement('embed');
 92     for (paramKey in flashParams) {
 93       if (flashParams.hasOwnProperty(paramKey)) {
 94         paramKey = (paramKey === 'movie') ? 'src' : paramKey;
 95         embed.setAttribute(paramKey, flashParams[paramKey]);
 96       }
 97     }
 98 
 99     embed.setAttribute('width', params.width);
100     embed.setAttribute('height', params.height);
101     embed.setAttribute('id', params.id);
102     embed.setAttribute('name', params.id);
103     embed.setAttribute('swLiveConnect', 'true');
104     embed.setAttribute('type', 'application/x-shockwave-flash');
105     element.appendChild(embed);
106     return element;
107   };

109   /**
110    * Called when the player is ready.
111    */
112   media.players.flash.prototype.onReady = function() {
113     var _this = this;
114     this.ready = true;
115     this.trigger('loadstart');
116 
117     // Perform a check for the duration every second until it shows up.
118     this.durationInterval = setInterval(function() {
119       if (_this.getDuration()) {
120         clearInterval(_this.durationInterval);
121         _this.trigger('durationchange', {duration: _this.getDuration()});
122       }
123     }, 1000);
124   };
125 
126   /**
127    * Should be called when the media is playing.
128    */
129   media.players.flash.prototype.onPlaying = function() {
130     var _this = this;
131     this.trigger('playing');
132     this.mediaInterval = setInterval(function() {
133       var cTime = _this.getCurrentTime();
134       var dur = _this.getDuration();
135       var data = {currentTime: cTime, duration: dur};
136       _this.trigger('timeupdate', data);
137     }, 1000);
138   };
139 
140   /**
141    * Should be called when the media is paused.
142    */
143   media.players.flash.prototype.onPaused = function() {
144     this.trigger('pause');
145     clearInterval(this.mediaInterval);
146   };
147 
148   /**
149    * Should be called when the meta data has finished loading.
150    */
151   media.players.flash.prototype.onMeta = function() {
152     clearInterval(this.durationInterval);
153     this.trigger('loadeddata');
154     this.trigger('loadedmetadata');
155     this.trigger('durationchange', {duration: this.getDuration()});
156   };
157

189   /**
190    * @see media.players.base#getPlayer
191    */
192   media.players.flash.prototype.getPlayer = function() {
193     // IE needs the object, everyone else just needs embed.
194     var object = $.browser.msie ? 'object' : 'embed';
195     return $(object, this.display).eq(0)[0];

204   };

206   /**
207    * Return the player time duration.
208    *
209    * @return {int} The player duration.
210    */
211   media.players.flash.prototype.getPlayerDuration = function() {
212     return 0;
213   };
214 
215   /**
216    * @see media.players.base#getDuration
217    */
218   media.players.flash.prototype.getDuration = function() {
219 
220     // Make sure to cache the duration since it is called often.
221     if (this.duration) {
222       return this.duration;
223     }
224     else if (this.isReady()) {
225       this.duration = this.getPlayerDuration();
226       return this.duration;
227     }
228     else {
229       return media.players.base.prototype.getDuration.call(this);
230     }
231   };
232 
233   /**
234    * @see media.players.base#isReady
235    */
236   media.players.flash.prototype.isReady = function() {
237     return (this.player && this.ready);
238   };
