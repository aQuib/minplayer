1 /** The Drupal namespace. */
  2 Drupal.media = Drupal.media || {};
  3 (function($, media) {
  4 
  5   /** All the media player implementations */
  6   media.players = media.players || {};
  7 
  8   /**
  9    * @class The Flash media player class to control the flash fallback.
 10    *
 11    * @extends media.display
 12    * @param {object} context The jQuery context.
 13    * @param {object} options This components options.
 14    */
 15   media.players.minplayer = function(context, options, mediaFile) {
 16 
 17     // Called when the flash provides a media update.
 18     this.onMediaUpdate = function(eventType) {
 19       if (this.ready) {
 20         switch (eventType) {
 21           case 'mediaMeta':
 22             media.players.flash.prototype.onMeta.call(this);
 23             break;
 24           case 'mediaPlaying':
 25             media.players.flash.prototype.onPlaying.call(this);
 26             break;
 27           case 'mediaPaused':
 28             media.players.flash.prototype.onPaused.call(this);
 29             break;
 30         }
 31       }
 32     };
 33 
 34     // Derive from players flash.
 35     media.players.flash.call(this, context, options, mediaFile);
 36   };
 37 
 38   window.onFlashPlayerReady = function(id) {
 39     if (media.player[id]) {
 40       media.player[id].media.onReady();
 41     }
 42   };
 43 
 44   window.onFlashPlayerUpdate = function(id, eventType) {
 45     if (media.player[id]) {
 46       media.player[id].media.onMediaUpdate(eventType);
 47     }
 48   };
 49 
 50   var debugConsole = console || {log: function(data) {}};
 51   window.onFlashPlayerDebug = function(debug) {
 52     debugConsole.log(debug);
 53   };
 54 
 55   /**
 56    * @see media.players.base#getPriority
 57    */
 58   media.players.minplayer.getPriority = function() {
 59     return 1;
 60   };
 61 
 62   /**
 63    * @see media.players.base#canPlay
 64    */
 65   media.players.minplayer.canPlay = function(file) {
 66     switch (file.mimetype) {
 67       case 'video/mp4':
 68       case 'video/x-webm':
 69       case 'video/quicktime':
 70       case 'video/3gpp2':
 71       case 'video/3gpp':
 72       case 'application/x-shockwave-flash':
 73       case 'audio/mpeg':
 74       case 'audio/mp4':
 75       case 'audio/aac':
 76       case 'audio/vnd.wave':
 77       case 'audio/x-ms-wma':
 78         return true;
 79 
 80       default:
 81         return false;
 82     }
 83   };
 84 
 85   // Define the prototype.
 86   media.players.minplayer.prototype = new media.players.flash();
 87   media.players.minplayer.prototype.constructor = media.players.minplayer;
 88 
 89   /**
 90    * @see media.players.base#create
 91    */
 92   media.players.minplayer.prototype.create = function() {
 93 
 94     media.players.base.prototype.flash.call(this);
 95 
 96     // The flash variables for this flash player.
 97     var flashVars = {
 98       'id': this.options.id,
 99       'debug': this.options.settings.debug,
100       'config': 'nocontrols',
101       'file': this.mediaFile.path,
102       'autostart': this.options.settings.autoplay
103     };
104 
105     // Return a flash media player object.
106     return media.players.minplayer.getFlash({
107       swf: this.options.swfplayer,
108       id: this.options.id + '_player',
109       playerType: 'flash',
110       width: this.options.settings.width,
111       height: '100%',
112       flashvars: flashVars,
113       wmode: this.options.wmode
114     });
115   };
116 
117   /**
118    * @see media.players.base#load
119    */
120   media.players.minplayer.prototype.load = function(file) {
121     media.players.flash.prototype.load.call(this, file);
122     if (this.isReady()) {
123       this.player.loadMedia(this.mediaFile.path, this.mediaFile.stream);
124     }
125   };
126 
127   /**
128    * @see media.players.base#play
129    */
130   media.players.minplayer.prototype.play = function() {
131     media.players.flash.prototype.play.call(this);
132     if (this.isReady()) {
133       this.player.playMedia();
134     }
135   };
136 
137   /**
138    * @see media.players.base#pause
139    */
140   media.players.minplayer.prototype.pause = function() {
141     media.players.flash.prototype.pause.call(this);
142     if (this.isReady()) {
143       this.player.pauseMedia();
144     }
145   };
146 
147   /**
148    * @see media.players.base#stop
149    */
150   media.players.minplayer.prototype.stop = function() {
151     media.players.flash.prototype.stop.call(this);
152     if (this.isReady()) {
153       this.player.stopMedia();
154     }
155   };
156 
157   /**
158    * @see media.players.base#seek
159    */
160   media.players.minplayer.prototype.seek = function(pos) {
161     media.players.flash.prototype.seek.call(this, pos);
162     if (this.isReady()) {
163       this.player.seekMedia(pos);
164     }
165   };
166 
167   /**
168    * @see media.players.base#setVolume
169    */
170   media.players.minplayer.prototype.setVolume = function(vol) {
171     media.players.flash.prototype.setVolume.call(this, vol);
172     if (this.isReady()) {
173       this.player.setVolume(vol);
174     }
175   };
176 
177   /**
178    * @see media.players.base#getVolume
179    */
180   media.players.minplayer.prototype.getVolume = function() {
181     if (this.isReady()) {
182       return this.player.getVolume();
183     }
184     else {
185       return media.players.flash.prototype.getVolume.call(this);
186     }
187   };
188 
189   /**
190    * @see media.players.flash#getPlayerDuration
191    */
192   media.players.minplayer.prototype.getPlayerDuration = function() {
193     return this.isReady() ? this.player.getDuration() : 0;
194   };
195 
196   /**
197    * @see media.players.base#getCurrentTime
198    */
199   media.players.minplayer.prototype.getCurrentTime = function() {
200     return this.isReady() ? this.player.getCurrentTime() : 0;
201   };
202 
203   /**
204    * @see media.players.base#getBytesLoaded
205    */
206   media.players.minplayer.prototype.getBytesLoaded = function() {
207     return this.isReady() ? this.player.getMediaBytesLoaded() : 0;
208   };
209 
210   /**
211    * @see media.players.base#getBytesTotal
212    */
213   media.players.minplayer.prototype.getBytesTotal = function() {
214     return this.isReady() ? this.player.getMediaBytesTotal() : 0;
215   };
216 }(jQuery, Drupal.media));
